162 private bool CompareTagByPreset(string message, out string[] handleMes)
163 {
164 handleMes = null;
165 string key = "", regex = "";
166
167 //筛选tag
168 //regex = (key = ShowTagsPreset.defaultShowTag) + ":";
169 if (
170 !ShowTagsPreset.showTags.Any(
171 (item) => {
172 regex = (key = item) + ":";
173 if (!item.EndsWith("Ban") && message.StartsWith(regex)) return true;
174 return false;
175 })
176 )
177 {
178 if (defaultTag)
179 {
180 handleMes = new string[] { "default", message };
181 return true;
182 }
183
184 return false;
185 }
186
187 handleMes = message.Split(regex, 2);
188 handleMes[0] = key;
189 return true;
